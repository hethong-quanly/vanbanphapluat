function getTime() {
  return new Date().toLocaleTimeString('vi-VN', { hour: '2-digit', minute: '2-digit' });

function escapeHTML(str) {
  return str
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

// Simple markdown-ish rendering for AI messages
function renderAIText(text) {
  let html = escapeHTML(text);

  // Bold **text** or *text*
  html = html.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
  html = html.replace(/\*(.+?)\*/g, '<em>$1</em>');

  // Code `inline`
  html = html.replace(/`([^`]+)`/g, '<code>$1</code>');

  // Lines starting with "  →" get arrow styling
  html = html.replace(/(→ .+)/g, '<span style="color:#C9A84C">$1</span>');

  // Numbered section headers "1." "2." "3." etc at start of line
  html = html.replace(/^(\d+\. .+)$/gm, '<strong style="color:#E8E0D0">$1</strong>');

  // Bullet lines "•"
  html = html.replace(/^(• .+)$/gm, '<span style="color:#B0A898">$1</span>');

  return html;
}

    const elapsed = Math.floor((Date.now() - sessionStart) / 1000);
    const m = String(Math.floor(elapsed / 60)).padStart(2, '0');
    const s = String(elapsed % 60).padStart(2, '0');
    $('sessionTime').textContent = `${m}:${s}`;
  }, 1000);
}

// ══════════════════════════════════════════════════════════════════
// COPY & SAVE
// ══════════════════════════════════════════════════════════════════
function copyMessage(msgId, btn) {
  const el = document.getElementById(msgId);
  if (!el) return;
  const text = el.innerText;
  navigator.clipboard.writeText(text).then(() => {
    btn.textContent = '✅ Đã sao chép';
    btn.classList.add('copied');
    setTimeout(() => {
      btn.textContent = '📋 Sao chép';
      btn.classList.remove('copied');
    }, 2000);
  }).catch(() => {
    // fallback
    const ta = document.createElement('textarea');
    ta.value = text; ta.style.position = 'fixed'; ta.style.opacity = '0';
    document.body.appendChild(ta); ta.select();
    document.execCommand('copy');
    document.body.removeChild(ta);
    showToast('✅ Đã sao chép vào clipboard');
  });
}

function saveMessage(msgId) {
  const el = document.getElementById(msgId);
  if (!el) return;
  const text = el.innerText;
  const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `LEXVIET_PhanTich_${new Date().toISOString().slice(0,10)}.txt`;
  a.click();
  URL.revokeObjectURL(url);
  showToast('💾 Đã lưu phân tích ra file');
}

let toastTimer;
function showToast(msg) {
  let t = document.querySelector('.toast');
  if (t) t.remove();

  t = document.createElement('div');
  t.className = 'toast';

  document.body.appendChild(t);
  toastTimer = setTimeout(() => { if (t.parentNode) t.remove(); }, 2500);
}

// ══════════════════════════════════════════════════════════════════
// SEND MESSAGE
// ══════════════════════════════════════════════════════════════════
async function sendMessage() {
  const input = $('userInput');
  const text = input.value.trim();
  if (!text || isLoading) return;
  if (text.length > 2000) { showToast('⚠️ Câu hỏi quá dài (tối đa 2000 ký tự)'); return; }

  // Reset input
  input.value = '';
  input.style.height = '';
  $('charCount').textContent = '0 / 2000';
  $('sendBtn').disabled = true;
  isLoading = true;

  // Update avatar
  $('aiAvatar').classList.add('loading');
  $('statusText').textContent = 'Đang phân tích...';

  // Append user message
  appendUserMessage(text);

  // Add to conversation history
  conversationHistory.push({ role: 'user', content: text });

  // Show typing
  showTyping();

  try {
    // Build messages array (limit context window to last 10 turns to avoid token overflow)
    const recentHistory = conversationHistory.slice(-10);

    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1500,
        system: SYSTEM_PROMPTS[currentMode] || SYSTEM_PROMPTS.general,
        messages: recentHistory
      })
    });

    removeTyping();

    if (!response.ok) {
      const errCode = response.status;
      if (errCode === 401) throw new Error('UNAUTHORIZED');
      if (errCode === 429) throw new Error('RATE_LIMIT');
      if (errCode === 529) throw new Error('OVERLOAD');
      throw new Error(`HTTP_${errCode}`);
    }

    const data = await response.json();
    const aiText = data.content?.map(c => c.type === 'text' ? c.text : '').join('') || 'Không có phản hồi.';

    // Add AI response to history
    conversationHistory.push({ role: 'assistant', content: aiText });

    appendAIMessage(aiText);

  } catch (err) {
    removeTyping();

    const errMsgMap = {
      'UNAUTHORIZED': '⚠️ Lỗi xác thực API (401). Vui lòng liên hệ quản trị viên cấu hình kết nối.\n\n---\n\n' + getMockResponse(text),
      'RATE_LIMIT':   '⏳ Hệ thống tạm thời quá tải. Đang dùng dữ liệu nội bộ:\n\n---\n\n' + getMockResponse(text),
      'OVERLOAD':     '⏳ Máy chủ AI đang bận. Đang dùng dữ liệu nội bộ:\n\n---\n\n' + getMockResponse(text),
    };

    const errKey = err.message;
    const fallbackText = errMsgMap[errKey] || getMockResponse(text);
    const isError = errKey === 'UNAUTHORIZED';

    conversationHistory.push({ role: 'assistant', content: fallbackText });
    appendAIMessage(fallbackText, isError);
  } finally {
    isLoading = false;
    $('sendBtn').disabled = false;
    $('aiAvatar').classList.remove('loading');
    $('statusText').textContent = 'Sẵn sàng phân tích nghiệp vụ';
    $('userInput').focus();
  }
}

}

// EXPORT CHAT
// ══════════════════════════════════════════════════════════════════
function exportChat() {
  if (conversationHistory.length === 0) { showToast('⚠️ Chưa có nội dung để xuất'); return; }

  const lines = [`LEXVIET – Lịch sử hội thoại\n`];
  lines.push(`Ngữ cảnh: ${currentModeLabel}`);
  lines.push(`Thời gian: ${new Date().toLocaleString('vi-VN')}`);
  lines.push('='.repeat(60) + '\n');

  conversationHistory.forEach(m => {
    const role = m.role === 'user' ? 'KIỂM SÁT VIÊN' : 'LEXVIET AI';
    lines.push(`[${role}]`);
    lines.push(m.content);
    lines.push('-'.repeat(40));
  });

  const blob = new Blob([lines.join('\n')], { type: 'text/plain;charset=utf-8' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `LEXVIET_HoiThoai_${new Date().toISOString().slice(0,10)}.txt`;
  a.click();
  URL.revokeObjectURL(url);
  showToast('📥 Đã xuất hội thoại ra file');
}

// INPUT EVENT LISTENERS
// ══════════════════════════════════════════════════════════════════
document.addEventListener('DOMContentLoaded', () => {
  const input = $('userInput');

  // Auto-resize textarea
  input.addEventListener('input', function () {
    this.style.height = '';
    this.style.height = Math.min(this.scrollHeight, 130) + 'px';
    const len = this.value.length;
    const cc = $('charCount');
    cc.textContent = `${len} / 2000`;
    cc.className = 'char-count' + (len > 1800 ? ' warn' : '');
    $('sendBtn').disabled = !this.value.trim() || isLoading;
  });

  // Enter to send
  input.addEventListener('keydown', function (e) {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      sendMessage();
    }
  });

  // Init send button disabled state
  $('sendBtn').disabled = true;
  input.addEventListener('input', () => {
    $('sendBtn').disabled = !input.value.trim() || isLoading;
  });

  // Start session timer
  startTimer();

  // Focus input
  input.focus();
});
</script>
